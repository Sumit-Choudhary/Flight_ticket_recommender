# SETUP NOTES
# ─────────────────────────────────────────────────────────────────────────────
# 1. Install Python deps:
#       pip install -r requirements.txt
#
# 2. Download Chromium browser binary (one-time, ~170 MB):
#       playwright install chromium
#
# 3. Required .env file in project root:
#       GEMINI_API_KEY=your_gemini_key
#       AMADEUS_CLIENT_ID=your_amadeus_client_id
#       AMADEUS_CLIENT_SECRET=your_amadeus_client_secret
#
# 4. Start the backend:
#       uvicorn main:app --reload --host 127.0.0.1 --port 8000
#
# 5. For Docker / Cloud Run — use the official Playwright base image to avoid
#    missing Chromium system dependencies:
#       mcr.microsoft.com/playwright/python:v1.49.0-jammy
#    Then in your Dockerfile:
#       RUN pip install -r requirements.txt
#       RUN playwright install chromium
# ─────────────────────────────────────────────────────────────────────────────
